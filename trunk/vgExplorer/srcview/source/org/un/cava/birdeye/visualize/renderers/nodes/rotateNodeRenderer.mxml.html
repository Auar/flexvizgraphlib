    <local:Rectangle id="rectangle" color="{this.data.data.@nodeColor}" nheight="10" nwidth="{this.data.data.@nodeSize}" toolTip="{this.data.data.@name}"/>

    </mx:VBox>
    
    <mx:Script>
        <![CDATA[
        import org.un.flex.graphLayout.visual.IVisualNode;
            
            public function setNodeRotation():void {
            
            if(this.data is IVisualNode) {
                this.rotation = (this.data as IVisualNode).orientAngle;
            }
            
            }
            
        ]]>
    </mx:Script>
